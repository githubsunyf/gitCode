/*苦涩红茶设计的有返回值的函数和递归练习-开发环境:Code::block*/
#include<stdio.h>

int imax(int a,int b);///ANSI C
int recur(int n);
long cloop(int n);
long rloop(int n);
void binary_calculate(unsigned long x);

int main(void)
{
    int i,j,f,test,num;
    unsigned long bnumber;
    printf("Please enter two integers to compare:\n");
    printf("1:");
    scanf("%d",&i);
    printf("2:");
    scanf("%d",&j);
    f = imax(i,j);///调用函数imax
    printf("The max number is %d\n",f);
    putchar('\n');

    printf("Please enter a number you like to have recur:\n");
    scanf("%d",&test);
    recur(test);///调用函数recur
    getchar();
    printf("\n\n");

    printf("Enter a value in the range 0-12  to calculate the factorials(a whatever char to quit):\n");
    while (scanf("%d", &num) == 1)
    {
        if (num < 0)
            printf("No negative numbers, please.\n");
        else if (num > 12)
            printf("Keep input under 13.\n");
        else
        {
            printf("Use loop: %d = %ld\n",
                   num, cloop(num));///调用函数cloop
            printf("Use recursion: %d = %ld\n",
                   num, rloop(num));///调用函数rloop
        }
        printf("Enter a value in the range 0-12 to calculate the factorials(a whatever char to quit):\n");
    }

    getchar();
    printf("\n\n");

    printf("Please enter a integer...(XXX(10)) to (XXX(2))_(a whatever char to quit)\n");
    while(scanf("%lu",&bnumber) == 1)
    {
        printf("The binary equipment:");
        binary_calculate(bnumber);///调用函数binary_calculate
        putchar('\n');
        printf("Please enter a integer...(XXX(10)) to (XXX(2))_(a whatever char to quit)\n");
    }
    return NULL;
}

/*定义函数*/
int imax(int a,int b)///比较两个数并返回最大值
{
    return(a > b ? a : b);
}

int recur(int n)///递归数值传回比较
{
    printf("The n:%d is in the location %p\n",n,&n);///#1
    if(n < 520)
    {
        recur(n + 1);///每级递归的变量n都属于本级递归私有
    }
    printf("The n:%d is in the location %p\n",n,&n);///#2
}

long cloop(int n)///循环实现阶乘算法
{
    long finally;

    for (finally = 1; n > 1; n--)
        finally *= n;

    return finally;
}

long rloop(int n)///递归实现阶乘算法
{
    long finally;

    if (n > 0)
        finally= n * rloop(n-1);
    else
        finally = 1;

    return finally;
}

void binary_calculate(unsigned long x)///递归实现十进制转二进制
{
    int y;

    y = x % 2;
    if(x >= 2)
        binary_calculate(x / 2);
    putchar(y == 0 ? '0' : '1');

    return;
}
